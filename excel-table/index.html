Excel-Table ,

An excel editor with Html5 features!

Overview:

The project just created for fun! Enjoy yourself!

Download from github.

Installation:

    <link rel="stylesheet" href="path/to/assets/iconfont.css">
    <link rel="stylesheet" href="path/to/excel-table.min.css">
    <script src="path/to/excel-table.min.js"></script>

Create table object:

    var table = new ExcelTable.Table();

Initial table object:

    table.init({
        target: '#table',
        data: {
            rows: 9,
            columns: 10,
            units: [
                {row: 0, column: 0, value: 6},
                {row: 0, column: 2, value: '=SUM(A0:B0)'},
                {row: 1, column: 1, value: 'fdd'},
                {row: 2, column: 2, value: '=function(){return 10;}'},
                {row: 2, column: 3, value: '=C2()'},
            ]
        }
    });

Create toolbar if you need:

    var toolbar = new ExcelTable.Toolbar();
    toolbar.init({
        target: '#table .excel-table-toolbar',
        items: [
            'import', 'export', '|',
            'undo', 'redo', '|',
            'cut', 'copy', 'paste', 'paste-transform', '|',
            'append-column', 'append-row', '|',
            'sort-asc', 'sort-desc', '|',
            'auto-fill'
        ]
    });
    toolbar.addTable(table);
